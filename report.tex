	\Cref{alg:nonsense} is really helpful for nothing.
	\begin{algorithm}
		\caption{\label{alg:nonsense} Predicting nonsense}
		\SetKwData{Left}{left}
		\SetKwData{This}{this}
		\SetKwData{Up}{up}
		\SetKwFunction{BlackBox}{BlackBox}
		\SetKwFunction{FindCompress}{FindCompress}
		\SetKwInOut{Input}{Input}
		\SetKwInOut{Output}{Output}
		\Input{A dataset $D\subset\mathbb{R}^{m\cdot n}$}
		\Output{ $y\in\{0, 1\}^k$}
		\For{$i\leftarrow 0$ \KwTo $k$}{
			get some random element $e\in D$\;
			$x$ \leftarrow \BlackBox{$e$}\;
			\While{$x\neq 1$}{$x$ \leftarrow \BlackBox{$e$}}
			$y[i] \leftarrow x$
		}
			\Return{$y$}
		
\end{algorithm}
